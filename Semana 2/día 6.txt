Ejercicio Clase día sábado

package inicio;
import java.util.Arrays;
import java.util.Scanner;
public class ClaseSabado {
	
	public static void main(String[] args) {
		int edad;
		int[] edades = new int[10];
		Scanner scan = new Scanner(System.in);
		
		
		for(int i = 0; i < edades.length; i++)
		{
			System.out.println("ingrese la edad");
			edad = scan.nextInt();
			if(edad>=18 && edad<=110)
			{
				edades[i] = edad;
			}else
			{
				i--;
			}
		}
		scan.close();
		Arrays.sort(edades);
		System.out.println("Edades mayores de 18 ingresadas en el arrego");
		for(int i = 0; i < edades.length; i++)
		{
		    System.out.println("edad " + i + 
                    " : "+ edades[i]);   
		}
		}
		         
}

Reflexión:

En esta clase aprendimos a usar los arreglos junto a un ciclo for, cómo en los ejercicios anteriores todo lo mostrado era
diferente pero aún asi llegaban al mismo resultado.
